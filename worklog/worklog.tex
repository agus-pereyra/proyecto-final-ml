\subsection{Sitios Webs}

\begin{itemize}
    \item \href{https://paperswithcode.com/datasets?modality=tabular}{Papers With Code – Datasets} — Datasets tabulares con papers y métricas de referencia asociadas.
    \item \href{https://www.openml.org/search?type=data&sort=runs&status=active&id=1489}{OpenML} — Repositorio de datasets para benchmarking, con resultados comparativos públicos.
    \item \href{https://www.kaggle.com/datasets}{Kaggle Datasets} — Datasets de competencias con notebooks y baselines de referencia.
    \item \href{https://archive.ics.uci.edu/}{UCI ML Repository} — Colección clásica de datasets para investigación en ML.
    \item \href{https://huggingface.co/datasets}{Hugging Face Datasets} — Repositorio creciente con datasets vinculados a papers recientes.
    % Ciencias naturales y medio ambiente
    \item \href{https://earthdata.nasa.gov/}{NASA Earthdata} — Datos climáticos, atmosféricos y oceánicos de misiones de la NASA.
    \item \href{https://www.ncei.noaa.gov/}{NOAA Climate Data} — Datos meteorológicos y climáticos históricos reales.
    \item \href{https://opendata.cern.ch/}{CERN Open Data Portal} — Datos de física de partículas del acelerador LHC.
    \item \href{https://www.pangaea.de/}{PANGAEA} — Datos de ciencias del mar y la tierra asociados a publicaciones científicas.
    % Salud y biología
    \item \href{https://physionet.org/}{PhysioNet} — Datos clínicos reales con papers asociados (UCI hospitalaria, señales fisiológicas).
    \item \href{https://gdc.cancer.gov/}{NIH GDC} — Datos genómicos y clínicos de investigación oncológica.
    \item \href{https://www.cdc.gov/brfss/}{CDC BRFSS} — Encuestas de salud poblacional de EE.UU., ampliamente usadas en ML clínico.
    \item \href{https://www.who.int/data/gho}{Global Health Observatory – OMS} — Indicadores de salud pública por país.
    % Ciencias sociales y economía
    \item \href{https://ipums.org/}{IPUMS} — Microdatos censales de todo el mundo, incluyendo América Latina.
    \item \href{https://ourworldindata.org/}{Our World in Data} — Datos globales sobre pobreza, energía, educación y salud con papers detrás.
    \item \href{https://data.worldbank.org/}{World Bank Open Data} — Indicadores económicos y sociales por país.
    % Gobierno y datos abiertos
    \item \href{https://data.gov/}{Data.gov} — Más de 290k datasets de agencias federales de EE.UU.
    \item \href{https://datos.gob.ar/}{datos.gob.ar} — Portal de datos abiertos del gobierno argentino.
    \item \href{https://data.europa.eu/}{EU Open Data Portal} — Datos abiertos de la Unión Europea.
    % Repositorios generales de investigación
    \item \href{https://zenodo.org/}{Zenodo} — Repositorio del CERN: cada dataset tiene DOI y referencia directa a su paper.
    \item \href{https://figshare.com/}{Figshare} — Datos publicados junto a artículos científicos de múltiples disciplinas.
    \item \href{https://datadryad.org/}{Dryad} — Datasets de papers de ciencias biológicas y ambientales.
    \item \href{https://dataverse.harvard.edu/}{Harvard Dataverse} — Repositorio de Harvard con datos de investigación de todas las disciplinas.
\end{itemize}
